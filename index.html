Business Case 1: Traffic Signal Control

Using Queue data structure

Description: Implement intelligent traffic signal systems that adapt to real-time traffic flow, reducing congestion and travel delays.

  • Regulates signal timing to minimize delays and maximize traffic movement efficiency.

  • Reduces accident risks by controlling vehicle and pedestrian interactions at intersections.

  • Allows priority for emergency vehicles, public transit, or pedestrians when needed.

  • Adjusts signal timing dynamically based on real-time traffic conditions.

Implementation for Business Case 1


#include <iostream>
#include <queue>
#include <string>
#include <thread>
#include <chrono>

using namespace std;

class TrafficSignalControl {
private:
    queue<int> northQueue, southQueue, eastQueue, westQueue;
    int greenTime = 3;

public:
    void addVehicleToQueue(int direction) {
        if (direction == 0) northQueue.push(1);
        if (direction == 1) southQueue.push(1);
        if (direction == 2) eastQueue.push(1);
        if (direction == 3) westQueue.push(1);
    }
    void processTraffic() {
        while (true) {
            processDirection(northQueue, "North");
            processDirection(southQueue, "South");
            processDirection(eastQueue, "East");
            processDirection(westQueue, "West");
            this_thread::sleep_for(chrono::seconds(1));
        }
    }

private:
    void processDirection(queue<int>& q, string direction) {
        if (!q.empty()) {
            cout << direction << " green for " << greenTime << " seconds." << endl;
            for (int i = 0; i < greenTime && !q.empty(); ++i) {
                q.pop();
                cout << "Vehicle passed through " << direction << " signal." << endl;
            }
        } else {
            cout << "No vehicles at " << direction << " signal." << endl;
        }
    }
};

int main() {
    TrafficSignalControl tsc;

    tsc.addVehicleToQueue(0);
    tsc.addVehicleToQueue(0);
    tsc.addVehicleToQueue(1);
    tsc.addVehicleToQueue(2);
    tsc.addVehicleToQueue(3);

    tsc.processTraffic();

    return 0;
}

Business Case 2: Transportation of Trash

Using travelling salesman problem

Description: Optimize trash transportation routes to minimize distance and cost, enabling efficient waste management.

  • Collection: Gather waste from residential, commercial, and industrial sources.

  • Sorting: Separate recyclable, hazardous, and organic materials for proper processing.

  • Transportation: Safely move waste to disposal, recycling, or treatment facilities

  • Disposal: Ensure environmentally safe disposal in landfills or incinerators.

Implementation for Business Case 2

#include iostream>
#include vector>
#include cmath>
#include climits>

using namespace std;

int distance(int x1, int y1, int x2, int y2) {
    return abs(x2 - x1) + abs(y2 - y1);
}

int tsp(int current, int visited, const vector>& dist, vector>& dp) {
    if (visited == (1 << dist.size()) - 1) {
        return dist[current][0];
    }

    if (dp[current][visited] != -1) {
        return dp[current][visited];
    }

    int ans = INT_MAX;
    for (int next = 0; next < dist.size(); ++next) {
        if ((visited & (1 << next)) == 0) {
            int newCost = dist[current][next] + tsp(next, visited | (1 << next), dist, dp);
            ans = min(ans, newCost);
        }
    }

    dp[current][visited] = ans;
    return ans;
}

int main() {
    vector> locations = {{0, 0}, {10, 2}, {5, 5}, {8, 8}, {3, 10}};
    vector trashAmounts = {5, 2, 8, 3, 6};

    int n = locations.size();
    
    vector> dist(n, vector(n, 0));
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n; ++j) {
            dist[i][j] = distance(locations[i].first, locations[i].second, locations[j].first, locations[j].second);
        }
    }

    vector> dp(n, vector((1 << n), -1));

    int minCost = tsp(0, 1, dist, dp);

    cout << "Minimum transportation cost (TSP): " << minCost << endl;

    return 0;
}
